#include <iostream>
using namespace std;

struct Calculator {
    float num1;               
    float num2;              

    
    float add() {
        return num1 + num2;
    }

  
    float subtract() {
        return num1 - num2;
    }

   
    float multiply() {
        return num1 * num2;
    }

    
    float divide() {
        if (num2 != 0) {
            return num1 / num2;
        } else {
            cout << "Error: divide by zero!" << endl;
            return 0; 
        }
    }
};

int main() {
    Calculator calc;
    char operation;
    char continueCalc;

    do {
        
        cout << "Please enter two numbers: ";
        cin >> calc.num1 >> calc.num2;

        
        cout << "Select the operation (+, -, *, /): ";
        cin >> operation;

       
        switch (operation) {
            case '+':
                cout << "the result: " << calc.add() << endl;
                break;
            case '-':
                cout << "the result: " << calc.subtract() << endl;
                break;
            case '*':
                cout << "the result: " << calc.multiply() << endl;
                break;
            case '/':
                cout << "the result: " << calc.divide() << endl;
                break;
            default:
                cout << "Invalid operation!" << endl;
        }

        
        cout << "Do you want to do another calculation? (y/n): ";
        cin >> continueCalc;
    } while (continueCalc == 'y' || continueCalc == 'Y');

    return 0;
}
